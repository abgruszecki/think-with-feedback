import math


def main(input_stream, output_stream):
    a, m = map(int, input_stream.readline().rstrip("\n").split())

    r0 = a % m
    if r0 == 0:
        print("Yes", file=output_stream)
    else:
        d = math.gcd(r0, m)
        m_prime = m // d
        if (m_prime & (m_prime - 1)) == 0:
            print("Yes", file=output_stream)
        else:
            print("No", file=output_stream)
